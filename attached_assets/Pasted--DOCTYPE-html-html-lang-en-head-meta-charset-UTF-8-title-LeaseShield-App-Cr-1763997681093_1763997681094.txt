  <script>
    // Simple screen switching
    function showScreen(screen) {
      document.getElementById('screen-home').classList.add('hidden');
      document.getElementById('screen-learn').classList.add('hidden');
      document.getElementById('screen-ask').classList.add('hidden');

      if (screen === 'home') {
        document.getElementById('screen-home').classList.remove('hidden');
      } else if (screen === 'learn') {
        document.getElementById('screen-learn').classList.remove('hidden');
      } else if (screen === 'ask') {
        document.getElementById('screen-ask').classList.remove('hidden');
      }
    }

    // Very simple front-end explanations (no AI yet)
    function handleExplain() {
      const input = (document.getElementById('user-question').value || '').trim().toLowerCase();
      const answerBox = document.getElementById('answer-box');
      const answerText = document.getElementById('answer-text');

      if (!input) {
        answerText.textContent = 'Please type a word or phrase from your report first.';
        answerBox.classList.remove('hidden');
        return;
      }

      let explanation = '';

      if (input.includes('charge-off') || input.includes('charge off')) {
        explanation = 'A charge-off is a debt that the lender has decided it is unlikely to collect. The account may be closed, but it can still appear on your credit report and hurt your score.';
      } else if (input.includes('collection')) {
        explanation = 'A collection is a bill that was not paid and has been sent to a collection agency. It usually means the account was seriously past due.';
      } else if (input.includes('utilization') || input.includes('utilisation')) {
        explanation = 'Utilization is how much of your available credit you are using. For example, if you have a $1,000 limit and owe $500, your utilization is 50%. Lower utilization is generally better.';
      } else if (input.includes('inquiry') || input.includes('inquiries') || input.includes('enquiry')) {
        explanation = 'An inquiry is a record that someone checked your credit report. For example, a bank might make an inquiry when you apply for a loan or credit card.';
      } else if (input.includes('30 days late') || input.includes('60 days late') || input.includes('90 days late')) {
        explanation = 'This means a payment was not received by the due date and was late by that many days. More days late usually has a bigger negative impact on your credit.';
      } else if (input.includes('delinquent') || input.includes('delinquency')) {
        explanation = 'Delinquent means a payment was missed or not made on time. A delinquent account is past due.';
      } else if (input.includes('past due')) {
        explanation = 'Past due means the payment should have been made already but has not been paid yet.';
      } else if (input.includes('default')) {
        explanation = 'Default means a serious failure to repay a debt as agreed, often after several missed payments.';
      } else {
        // Generic fallback explanation
        explanation = 'This looks like a specific term from your credit report. In general, check whether it refers to: (1) a type of account (loan or card), (2) a status (paid, late, collection), or (3) a number (balance, limit, or date). If you are unsure, you may want to speak with a financial professional for personal advice.';
      }

      answerText.textContent = explanation;
      answerBox.classList.remove('hidden');
    }

    // Start at home screen
    showScreen('home');
  </script>
