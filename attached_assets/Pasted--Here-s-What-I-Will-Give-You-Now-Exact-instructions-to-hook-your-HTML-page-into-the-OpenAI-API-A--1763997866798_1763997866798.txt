import express from "express";
import OpenAI from "openai";
import cors from "cors";

const app = express();
app.use(express.json());
app.use(cors());

// 1. Initialize your OpenAI API client
const client = new OpenAI({
  apiKey: process.env.OPENAI_API_KEY
});

// 2. POST route the frontend will call
app.post("/explain", async (req, res) => {
  let term = (req.body.term || "").toLowerCase();

  // 3. Basic safety filters (no SSNs, account numbers, etc.)
  if (term.match(/\b\d{3}-\d{2}-\d{4}\b/) || term.includes("ssn")) {
    return res.json({
      explanation:
        "For your safety, please do not enter Social Security numbers or personal identifiers."
    });
  }

  if (/\d{8,}/.test(term)) {
    return res.json({
      explanation:
        "This looks like an account or loan number. For privacy reasons, please enter only the WORD or PHRASE you'd like explained."
    });
  }

  try {
    // 4. Call OpenAI to explain the term
    const response = await client.chat.completions.create({
      model: "gpt-4o-mini",
      messages: [
        {
          role: "system",
          content:
            "You are a simple, friendly assistant that explains credit report terms in very plain English. Keep explanations short. Do not give legal advice."
        },
        {
          role: "user",
          content: `Explain the credit report term: "${term}".`
        }
      ]
    });

    const explanation = response.choices[0].message.content;

    res.json({ explanation });
  } catch (error) {
    console.error(error);
    res.status(500).json({
      explanation: "Sorry, something went wrong. Please try again."
    });
  }
});

// Start server
app.listen(3000, () => {
  console.log("LeaseShield API running on port 3000");
});

  try {

      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ term: input })
    });

    const data = await response.json();
    answerText.textContent = data.explanation || "No answer received.";
    answerBox.classList.remove('hidden');

  } catch (err) {
    answerText.textContent = "Something went wrong. Please try again.";
    answerBox.classList.remove('hidden');
  }
}
