7) Replit snippet to enforce accepted file types (Node + multer)

Drop this into your upload handler to block risky formats:

const allowedExt = new Set([".pdf", ".jpg", ".jpeg", ".png"]);

function fileFilter(req, file, cb) {
  const ext = (require("path").extname(file.originalname) || "").toLowerCase();
  if (!allowedExt.has(ext)) return cb(new Error("Unsupported file type."), false);
  cb(null, true);
}
