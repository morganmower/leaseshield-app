A) Replit Object Storage helper (Node/TS)

Use the sidecar proxy (127.0.0.1:1106) exactly like you described.

// src/lib/objstore.ts
import { Storage } from "@google-cloud/storage";

const apiEndpoint = "http://127.0.0.1:1106";
export const BUCKET = process.env.REPLIT_OBJSTORE_BUCKET!; 
// e.g. replit-objstore-38df3fee-63f6-49b7-8f77-81e0e71280fd

export const storage = new Storage({ apiEndpoint });
export const bucket = storage.bucket(BUCKET);

export function makeDbPath(objectKey: string) {
  // objectKey like ".private/applicants/<uuid>.ext"
  return `/${BUCKET}/${objectKey}`;
}

export function parseDbPath(dbPath: string) {
  // "/bucket/.private/applicants/x.pdf"
  const parts = dbPath.split("/").filter(Boolean);
  const bucketName = parts[0];
  const objectKey = parts.slice(1).join("/");
  return { bucketName, objectKey };
}

B) Upload route: switch to memoryStorage + upload to object storage

No diskStorage. No writing into uploads/.

// src/routes/uploads.ts
import multer from "multer";
import path from "path";
import { randomUUID } from "crypto";
import { bucket, makeDbPath } from "../lib/objstore";

// memory only
const upload = multer({
  storage: multer.memoryStorage(),
  limits: { fileSize: 10 * 1024 * 1024 },
});

export const applicantUploadHandler = [
  upload.single("file"),
  async (req: any, res: any) => {
    const file = req.file;
    if (!file) return res.status(400).json({ error: "No file uploaded" });

    const ext = path.extname(file.originalname).toLowerCase() || "";
    const id = randomUUID();
    const filename = `${id}${ext}`;

    const objectKey = `.private/applicants/${filename}`;
    const dbPath = makeDbPath(objectKey);

    // Upload buffer to object storage
    const blob = bucket.file(objectKey);
    await blob.save(file.buffer, {
      resumable: false,
      contentType: file.mimetype,
      metadata: {
        metadata: {
          originalName: file.originalname,
          // add personId/fileType/etc if you want
        },
      },
    });

    // Write DB row (example — adapt to your query layer)

C) Download route: unified resolver (object storage first)

This is where you stop “ghost 404s” from legacy paths.

// src/routes/files.ts
import fs from "fs";
import path from "path";
import { bucket, parseDbPath, BUCKET } from "../lib/objstore";

export async function serveFile(req: any, res: any) {
  // fetch db row by id
  // const row = await db.query("select stored_path, mime_type, original_name, availability_status from rental_submission_files where id=$1", [req.params.id]);
  const row = req.fileRow; // placeholder

  const storedPath = row.stored_path as string;

  // New object-storage paths
  if (storedPath.startsWith(`/${BUCKET}/`)) {
    const { objectKey } = parseDbPath(storedPath);
    const f = bucket.file(objectKey);

    const [exists] = await f.exists();
    if (!exists) {
      // optionally update DB: availability_status='missing'
      return res.status(404).json({ error: "File unavailable (missing from storage)" });
    }

    res.setHeader("Content-Type", row.mime_type || "application/octet-stream");
    res.setHeader("Content-Disposition", `inline; filename="${row.original_name || "file"}"`);

    f.createReadStream().pipe(res);
    return;
  }

  // Legacy local disk paths (read-only / best-effort)
  if (storedPath.startsWith("uploads/")) {
    const localPath = path.join(process.cwd(), storedPath);
    if (!fs.existsSync(localPath)) {
      // update DB to missing to stop retries
      return res.status(404).json({ error: "File unavailable (legacy local file missing)" });
    }
    res.setHeader("Content-Type", row.mime_type || "application/octet-stream");
    fs.createReadStream(localPath).pipe(res);
    return;
  }

  return res.status(400).json({ error: "Unrecognized stored_path format" });
}
