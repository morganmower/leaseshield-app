In your index.html (or whatever page is your marketing site), add this where you want the chat to appear (bottom-right popup style):

<div id="ls-chat-widget">
  <div id="ls-chat-header">LeaseShield Assistant</div>
  <div id="ls-chat-messages"></div>
  <form id="ls-chat-form">
    <input
      id="ls-chat-input"
      type="text"
      placeholder="Ask a question about leases or LeaseShield Pro..."
      autocomplete="off"
      required
    />
    <button type="submit">Send</button>
  </form>
</div>

Then add this JS at the bottom of your HTML (before </body>):

<script>
  const form = document.getElementById("ls-chat-form");
  const input = document.getElementById("ls-chat-input");
  const messagesDiv = document.getElementById("ls-chat-messages");

  function addMessage(text, type) {
    const div = document.createElement("div");
    div.classList.add("ls-msg");
    div.classList.add(type === "user" ? "ls-msg-user" : "ls-msg-bot");
    div.textContent = text;
    messagesDiv.appendChild(div);
    messagesDiv.scrollTop = messagesDiv.scrollHeight;
  }

  form.addEventListener("submit", async (e) => {
    e.preventDefault();
    const text = input.value.trim();
    if (!text) return;

    addMessage(text, "user");
    input.value = "";

    addMessage("Thinking...", "bot");

    try {
      const res = await fetch("/api/chat", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ message: text }),
      });

      const data = await res.json();
      // remove the "Thinking..." line
      messagesDiv.lastChild.remove();
      addMessage(data.reply || "Sorry, I had trouble answering that.", "bot");
    } catch (err) {
      console.error(err);
      messagesDiv.lastChild.remove();
      addMessage("Error talking to LeaseShield Assistant. Please try again.", "bot");
    }
  });
</script>


Now when you run your Replit app and visit the site, you should see a little LeaseShield Assistant box in the corner that talks using the same brain as your custom prompt.
