A) Object storage helper (reuses your bucket + sidecar)
// src/lib/objstore.ts
import { Storage } from "@google-cloud/storage";

export const OBJSTORE_ENDPOINT = "http://127.0.0.1:1106";
export const BUCKET = process.env.REPLIT_OBJSTORE_BUCKET!;

export const storage = new Storage({ apiEndpoint: OBJSTORE_ENDPOINT });
export const bucket = storage.bucket(BUCKET);

export function toDbPath(objectKey: string) {
  // objectKey: ".private/applicants/<uuid>.ext"
  return `/${BUCKET}/${objectKey}`;
}

export function isObjstorePath(p: string) {
  return p.startsWith(`/${BUCKET}/.private/`);
}

export function dbPathToObjectKey(p: string) {
  // "/bucket/.private/applicants/x.pdf" -> ".private/applicants/x.pdf"
  const parts = p.split("/").filter(Boolean);
  return parts.slice(1).join("/"); // ".private/applicants/x.pdf"
}

  async (req: any, res: any) => {
    const file = req.file;
    if (!file) return res.status(400).json({ error: "No file uploaded" });

    const ext = path.extname(file.originalname).toLowerCase() || "";

C) Download route (inline exists() check + mark missing)
import fs from "fs";
import path from "path";
import { bucket, isObjstorePath, dbPathToObjectKey } from "../lib/objstore";

export async function downloadApplicantFile(req: any, res: any) {
  // const row = await db.query("SELECT ... FROM rental_submission_files WHERE id=$1", [req.params.id]);
  const row = req.fileRow; // placeholder

  const storedPath = row.stored_path as string;

  // Object Storage
  if (isObjstorePath(storedPath)) {
    const objectKey = dbPathToObjectKey(storedPath);
    const f = bucket.file(objectKey);

    const [exists] = await f.exists();
    if (!exists) {
      // await db.query("UPDATE rental_submission_files SET availability_status='missing' WHERE id=$1", [row.id]);
      return res.status(404).json({
        error: "File unavailable (missing from storage).",
        file_id: row.id,
      });
    }

    res.setHeader("Content-Type", row.mime_type || "application/octet-stream");
    res.setHeader(
      "Content-Disposition",
      `inline; filename="${row.original_name || "file"}"`
    );
    return f.createReadStream().pipe(res);
  }

  // Legacy Local (temporary support)
  if (storedPath.startsWith("uploads/")) {
    const localPath = path.join(process.cwd(), storedPath);
    if (!fs.existsSync(localPath)) {
      // await db.query("UPDATE rental_submission_files SET availability_status='missing' WHERE id=$1", [row.id]);
      return res.status(404).json({
        error: "File unavailable (legacy local file missing).",
        file_id: row.id,
      });
    }

    res.setHeader("Content-Type", row.mime_type || "application/octet-stream");
    return fs.createReadStream(localPath).pipe(res);
  }

  return res.status(400).json({ error: "Unrecognized stored_path format." });
}
