2. Value Gates (Progressive Unlock – Day 1 Basic, Day 3 Decoder, Day 5 Full)How: Use localStorage + date check to gate features (e.g., hide decoder until Day 3). Email nudges via Stripe customer emails (manual for now; automate with Zapier free tier).

import React, { useEffect, useState } from 'react';

// Gated Feature Component (use for decoder/ledger)
const GatedFeature = ({ dayRequired, featureName, children }) => {
  const [trialStart] = useState(() => localStorage.getItem('trialStart') || new Date().toISOString());
  const [unlocked, setUnlocked] = useState(false);

  useEffect(() => {
    const startDate = new Date(trialStart);
    const daysElapsed = Math.floor((new Date() - startDate) / (1000 * 60 * 60 * 24));
    if (daysElapsed >= dayRequired) {
      setUnlocked(true);
    }
  }, [trialStart, dayRequired]);

  if (!unlocked) {
    return (
      <div style={{padding: '2rem', textAlign: 'center', border: '2px dashed #10b981', borderRadius: '8px', background: '#f0f9ff'}}>
        <h3 style={{color: '#3182ce'}}>Unlock {featureName} on Day {dayRequired}</h3>
        <p>You're on Day {Math.floor((new Date() - new Date(localStorage.getItem('trialStart') || new Date())) / (1000 * 60 * 60 * 24)) + 1}. Keep using to unlock more!</p>
        <button onClick={() => localStorage.setItem('trialStart', new Date().toISOString())} style={{background: '#10b981', color: 'white', padding: '0.5rem 1rem', borderRadius: '4px', border: 'none'}}>
          Refresh Trial Start
        </button>
      </div>
    );
  }

  return <>{children}</>;
};

// Example Usage (wrap decoder section)

  </div>
</GatedFeature>

  <div className="full-tools">
    {/* Your ledger/comms JSX here */}
  </div>
</GatedFeature>

5. Test Paywall Alt (A/B 50% Traffic)How: JS randomizer on landing—50% see trial CTA, 50% see "Pay $10 first month" (no trial).

import React, { useState, useEffect } from 'react';

const ABTestWrapper = ({ children }) => {
  const [variant, setVariant] = useState(null);

  useEffect(() => {
    const stored = localStorage.getItem('ab-variant');
    if (!stored) {
      const rand = Math.random() < 0.5 ? 'trial' : 'paywall';
      localStorage.setItem('ab-variant', rand);
      setVariant(rand);
    } else {
      setVariant(stored);
    }
  }, []);

  if (variant === 'paywall') {
    return (
      <div style={{textAlign: 'center', margin: '2rem 0', padding: '1rem', background: '#dbeafe', borderRadius: '8px'}}>
        <p><strong>No trial today?</strong> Pay $10 first month, cancel anytime.</p>
        <a href="YOUR-PAYWALL-STRIPE-LINK" className="cta-button">Pay $10 Now – Full Access</a>
      </div>
    );
  }

  return children; // Show trial version
};

// Wrap hero/pricing
<ABTestWrapper>
  {/* Your hero and pricing JSX */}
</ABTestWrapper>
